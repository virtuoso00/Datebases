书店管理系统中的数据库模块

1.书店管理系统的简单介绍以及数据库中包含的表的内容：

书店管理系统调用数据库来存储租赁图书用户的信息，图书的信息以及注册的用户信息。当然此书店管理系统还会记录读者对于一本树各个方面的评价星级，以便于用户搜索某一类型的图书时可以由系统推荐评价星级较高的图书，以便读者租赁和购买。一个简单的书店管理系统包括书店内书籍的信息、注册用户的信息、用户的租赁信息以及用户购买图书的信息。此系统功能分为面向注册用户、面向管理员和面向书店派送员三部分，其中用户可以进行租赁、续租、归还、购买书籍以及查询书籍等操作，管理员可以完成书籍和用户的增加，删除和修改以及对用户租赁、续租、归还、购买的确认、快递公司的派送员可以查看到用户购买和租赁书籍的订单信息并从总中获取用户的姓名、联系电话、以及住址。

2. 需求分析：

针对一般书店管理信息系统的需求，通过对书店管理工作过程的内容和数据流程分析，设计如下面所示的数据项：

注册用户信息

属性：用户ID，用户姓名，用户性别，用户电话，用户住址，生效日期，失效日期，违章状况，累计租赁书籍，备注

主键：用户ID

书籍信息

属性：ISBN，书名，书籍类型，书籍是否有包装，作者，出版社，出版日期，简介，购买价格，租赁价格，备注

主键：ISBN

管理员信息

属性：工作号，姓名，型别，电话，家庭住址，备注

主键：工作号

派送员信息

属性：派送员编号，姓名，型别，电话，家庭住址，备注

主键：派送员编号

4.逻辑设计

从理论‘E/R模型’到理论‘关系模型’的整理转换，通过E/R模型到关系模型的转化，可以得到如下关系模式：

属性：工作号，ISBN，添加时间，是否在店。

管理员用户关系

属性：工作号，用户ID，确认租还。

订单信息

属性：订单号，用户ID，ISBN，派送员编号，用户姓名，书籍名称，派送员姓名，派送员电话，书籍是否有包装，用户电话，用户住址，备注。

主键：订单号，派送员编号

6. 物理设计

从理论‘关系模型’到实现/实施‘数据库建立’，物理文件的安排和建立索引

6.1 建立索引

为了提高在表中搜索元组的速度，在实际实现的时候应该基于键码建立索引是各表中建立索引的表项：

  1. 用户信息（用户ID）
  2. 书籍信息（ISBN，书籍ID）
  3. 管理员信息（工作号）
  4. 派送员信息（派送员编号）
  5. 租赁（工作号，读者学号，书籍ID）
  6. 购买书籍订单（订单号）
  7. 租赁书籍订单（订单号）
  8. 管理员_书籍（工作号，ISBN）
  9. 管理员_用户（工作号，用户ID）
  10. 用户_派送员（用户ID，派送员编号）

6.2 用SQL实现设计

实现该设计的环境为windows10-MySQL 5.7
create database 书店管理;

6.2.1 建立出售图书信息表

create table 出售图书信息
  (
    ISBN varchar(20) not null primary key,
    书籍名称 varchar(50) not null,
    作者 varchar(12) not null,
    类型 varchar(20) not null,
    出版社 varchar(50),
    出版日期 datetime,
    简介 varchar(200),
    购买价格 numeric(5,2) not null,
    备注 varchar(100)
  );

6.2.2 建立租赁图书信息表

create table 租赁图书信息
  (
    书籍ID varchar(10) not null primary key,
    ISBN varchar(20) not null,
    租赁价格 int not null,
    是否被租约 varchar(2) not null
    foreign key(ISBN) references 出售图书
  );

6.2.3 建立用户信息表

create table 用户信息
  (
    用户ID varchar(10) not null primary key,
    用户姓名 varchar(10) not null,
    密码 varchar(16) not null,
    住址 varchar(100) not null,
    联系电话 varchar(12) not null,
    注册时间 datetime,
    违章情况 varchar(2),
    累计租赁书籍 int,
    备注 varchar(100)
  );

6.2.4 建立管理员信息表

create table 管理员信息
  (
    工号 varchar(12) not null primary key,
    管理员密码 varchar(16) not null,
    管理员姓名 varchar(12) not null,
    管理员电话 varchar(12) not null,
    备注 varchar(100)
  );

6.2.5 建立派送员信息表

create table 派送员信息
  (
    派送员编号 varchar(12) not null primary key,
    派送员密码 varchar(16) not null,
    派送员姓名 varchar(12) not null,
    派送员电话 varchar(12) not null,
    备注 varchar(100)
  );

6.2.6 建立租赁情况表

create table 租赁情况
  (
    工号 varchar(12) not null primary key,
    ISBN varchar(20) not null,
    用户ID varchar(10) not null,
    租赁日期 datetime not null,
    还书日期 datetime not null,
    备注 varchar(100)
    foreign key(工号) references 管理员信息,
    foreign key(ISBN) references 出售图书信息,
    foreign key(用户ID) references 用户信息
  );

6.2.7 建立购买订单信息表

create table 购买订单信息
  (
    订单号 varchar(20) not null,
    派送员编号 varchar(12) not null,
    用户ID varchar(10) not null,
    ISBN varchar(20) not null,
    书籍名称 varchar(50) not null,
    购买价格 numeric(5,2) not null,
    购买日期 datetime not null,
    用户姓名 varchar(10) not null,
    住址 varchar(100) not null,
    派送员电话 varchar(12) not null,
    备注 varchar(100),
    primary key(订单号,派送员编号,用户ID),
    foreign key(派送员编号) references 派送员信息,
    foreign key(用户ID) references 用户信息,
    foreign key(ISBN) references 出售图书信息
  );

6.2.8 建立租赁信息订单表

create table 租赁订单信息
  (
    订单号 varchar(20) not null,
    派送员编号 varchar(12) not null,
    用户ID varchar(10) not null,
    书籍ID varchar(10) not null,
    ISBN varchar(20) not null,
    书籍名称 varchar(50) not null,
    租赁价格 int not null,
    租赁时间 datetime not null,
    用户姓名 varchar(10) not null,
    住址 varchar(100) not null,
    派送员电话 varchar(12) not null,
    备注 varchar(100),
    primary key(订单号,派送员编号,用户ID,书籍ID),
    foreign key(派送员编号) references 派送员信息,
    foreign key(用户ID) references 用户信息,
    foreign key(书籍ID) references 租赁图书信息,
    foreign key(ISBN) references 出售图书信息
  );

6.2.9 建立书籍评价表

create table 书籍评价信息
  (
    ISBN varchar(20) not null primary key,
    书籍名称 varchar(50) not null,
    内容合理评分 int,
    学术性评分 int,
    语言简洁评分 int,
    foreign key(ISBN) references 出售图书信息
  );
